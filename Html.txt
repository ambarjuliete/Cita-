<script>
function moveButton() {
const noBtn = document.getElementById('no-btn');

// Calcula posiciones aleatorias dentro de la pantalla
const x = Math.random() * (window.innerWidth - noBtn.offsetWidth - 40);
const y = Math.random() * (window.innerHeight - noBtn.offsetHeight - 40);

// Mueve el botón a la nueva posición absoluta
noBtn.style.left = x + 'px';
noBtn.style.top = y + 'px';
}
